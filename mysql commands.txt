CREATE TABLE users (
    id INT AUTO_INCREMENT PRIMARY KEY,

    business_name VARCHAR(100) NOT NULL,
    owner_name VARCHAR(100) NOT NULL,
    email VARCHAR(120) NOT NULL UNIQUE,
    password VARCHAR(128) NOT NULL,

    msme_category VARCHAR(10) NOT NULL,  -- Micro / Small / Medium
    business_type VARCHAR(100) NOT NULL,
    sector VARCHAR(100) NOT NULL,

    industry_sector VARCHAR(100),
    annual_turnover FLOAT,
    location VARCHAR(150),
    contact_number VARCHAR(15),
    registration_number VARCHAR(50),

    created_at DATETIME DEFAULT CURRENT_TIMESTAMP
);

------------------------------------------------------------------Emmision Table ------------------------------------------------------------------------------------------


CREATE TABLE emissions (
    id INT AUTO_INCREMENT PRIMARY KEY,
    user_id INT NOT NULL,
    category VARCHAR(50) NOT NULL,       -- Electricity, Fuel, etc.
    sub_type VARCHAR(50) NOT NULL,       -- Diesel, Petrol, etc.
    value FLOAT NOT NULL,                -- User input quantity
    unit VARCHAR(20) NOT NULL,           -- kg, L, kWh, etc.
    emission FLOAT NOT NULL,             -- Calculated COâ‚‚ emission in kg
    date DATE NOT NULL,                  -- Date of activity

    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ON UPDATE CURRENT_TIMESTAMP,

    FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE
);
